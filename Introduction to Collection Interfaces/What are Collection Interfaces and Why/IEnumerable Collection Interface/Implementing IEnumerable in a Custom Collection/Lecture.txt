Implementing IEnumerable in a Custom Collection
Implementing IEnumerable or IEnumerable<T> in a custom collection allows it to be used with a foreach loop. Hereâ€™s how to implement IEnumerable<T> in a custom collection:

Example 1: Implementing IEnumerable<T> in a Custom Collection
This example demonstrates how to implement the IEnumerable<T> interface in a custom collection class, allowing it to be iterated over with a foreach loop.

using System;
using System.Collections;
using System.Collections.Generic;


namespace CustomCollectionExample
{
    public class CustomCollection<T> : IEnumerable<T>
    {
        private List<T> items = new List<T>();


        public IEnumerator<T> GetEnumerator()
        {
            for (int i = 0; i < items.Count; i++)
            {
                yield return items[i];
            }
        }


        IEnumerator IEnumerable.GetEnumerator()
        {
            return GetEnumerator();
        }


        public void Add(T item)
        {
            items.Add(item);
        }
    }


    class Program
    {
        static void Main(string[] args)
        {
            var myCollection = new CustomCollection<int>();
            myCollection.Add(1);
            myCollection.Add(2);
            myCollection.Add(3);


            foreach (var item in myCollection)
            {
                Console.WriteLine(item);
            }
        Console.ReadKey();
        }
    }
}
